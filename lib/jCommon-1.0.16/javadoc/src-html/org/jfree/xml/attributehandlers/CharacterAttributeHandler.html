043    package org.jfree.xml.attributehandlers;
044    
045    /**
046     * A class that handles the conversion of {@link Character} attributes to and from an appropriate
047     * {@link String} representation.
048     */
049    public class CharacterAttributeHandler implements AttributeHandler {
050    
051        /**
052         * Creates a new attribute handler.
053         */
054        public CharacterAttributeHandler() {
055            super();
056        }
057    
058        /**
059         * Converts the attribute to a string.
060         * 
061         * @param o  the attribute ({@link Character} expected).
062         * 
063         * @return A string representing the {@link Character} value.
064         */
065        public String toAttributeValue(final Object o) {
066            final Character in = (Character) o;
067            return in.toString();
068        }
069    
070        /**
071         * Converts a string to a {@link Character}.
072         * 
073         * @param s  the string.
074         * 
075         * @return a {@link Character}.
076         */
077        public Object toPropertyValue(final String s) {
078            if (s.length() == 0) {
079                throw new RuntimeException("Ugly, no char set!");
080            }
081            return new Character(s.charAt(0));
082        }
083    }
