043    package org.jfree.ui;
044    
045    import java.awt.Graphics2D;
046    import java.awt.geom.Rectangle2D;
047    
048    /**
049     * An interface for an object that can draw itself within an area on a Graphics2D.
050     *
051     * @author David Gilbert
052     */
053    public interface Drawable {
054    
055        /**
056         * Draws the object.
057         *
058         * @param g2  the graphics device.
059         * @param area  the area inside which the object should be drawn.
060         */
061        public void draw(Graphics2D g2, Rectangle2D area);
062    
063    }
